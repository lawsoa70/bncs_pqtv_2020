<%@LANGUAGE="VBSCRIPT" CODEPAGE="1252"%> File Format - workstations.xml

Overview

workstations.xml contains the settings for each workstation

Each workstation element defines the name and default user of the workstation specified in the id attribute

Sample Contents - /config/workstations.xml

<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE workstations SYSTEM "workstations.dtd">
<workstations>
	<workstation id="11" name="desk1" user="desk"/>
	<workstation id="12" name="desk2" user="desk"/>
	<workstation id="13" name="desk3" user="desk"/>
	<workstation id="14" name="desk4" user="desk"/>
	<workstation id="21" name="cp" user="demo"/>
	<workstation id="22" name="news" user="demo"/>
	<workstation id="23" name="pres" user="demo"/>
	<workstation id="33" name="chris" user="desk"/>
	<workstation id="94" name="charlotte" user="desk"/>
	<workstation id="899" name="chris" user="desk"/>
</workstations>

DTD

<?xml version="1.0" encoding="UTF-8"?>
  <!ELEMENT workstations (workstation+)>

<!ELEMENT workstation EMPTY>
  <!ATTLIST workstation id CDATA #REQUIRED>
  <!ATTLIST workstation name CDATA #REQUIRED>
  <!ATTLIST workstation user CDATA #REQUIRED>
