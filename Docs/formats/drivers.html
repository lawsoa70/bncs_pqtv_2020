Overview

drivers.xml contains the list of drivers present within the system

 

Sample Contents 

<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE drivers SYSTEM "drivers.dtd">
<drivers>
	<driver id="100" name="IBC Combiner" type="combiner"/>
	<driver id="120" name="SDI Router" type="freeway"/>
	<driver id="191" name="CC Multiviewer" type="kaleido"/>
	<driver id="201" name="Vistek" type="dart"/>
	<driver id="230" name="RollCall" type="rolldrv"/>
	<driver id="240" name="Axon" type="synapse"/>
</drivers>

DTD

<?xml version="1.0" encoding="UTF-8"?>
  <!ELEMENT drivers (driver+)>

<!ELEMENT driver EMPTY>
  <!ATTLIST driver id CDATA #REQUIRED>
  <!ATTLIST driver name CDATA #REQUIRED>
  <!ATTLIST driver type CDATA #REQUIRED>
