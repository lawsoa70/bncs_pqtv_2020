DTD

<?xml version="1.0" encoding="UTF-8"?>

<!ELEMENT driverdescription (paramgroup+)>
	<!ATTLIST driverdescription id CDATA #REQUIRED>
	<!ATTLIST driverdescription type CDATA #REQUIRED>
	<!ATTLIST driverdescription deviceid CDATA #REQUIRED>

<!ELEMENT paramgroup (param+)>
	<!ATTLIST paramgroup id CDATA #REQUIRED>

<!ELEMENT param EMPTY>
	<!ATTLIST param id CDATA #REQUIRED>
	<!ATTLIST param type CDATA #REQUIRED>
	<!ATTLIST param default CDATA #REQUIRED>
