<%@LANGUAGE="VBSCRIPT" CODEPAGE="1252"%> File Format - applications.xml

Overview

typehandlers.xml defines the application that will be used to adjust a device type

Sample Contents - /config/typehandlers.xml

<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE typehandlers SYSTEM "typehandlers.dtd">
<typehandlers>
	<type id="al_chain" default="al_chain"/>
	<type id="al_sat" default="al_sat"/>
	<type id="alteia" default="alteiap"/>
	<type id="arc20" default="arc20p"/>
	<type id="iqdsig" default="iqdsigp"/>
	<type id="iqdsyn" default="iqdsynp"/>
	<type id="NDS3000" default="ND3000P"/>
	<type id="tt1260" default="tt1260p"/>
	<type id="V1642" default="V1642P"/>
	<type id="V1647" default="V1647P"/>
</typehandlers>

DTD

<?xml version="1.0" encoding="UTF-8"?>
<!ELEMENT typehandlers (type+)>
<!ELEMENT type EMPTY>
<!ATTLIST type
	id CDATA #REQUIRED
	default CDATA #REQUIRED
>
