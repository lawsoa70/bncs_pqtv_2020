Overview

The config/devicetype folder contains descriptions for each of the device types present within the system

The device contains parameter definitions of the following type:

Type Usage Comments
enum a known set of discrete values An enum can have an explicit state list or use a template
range values lie within a continuous range  
list a list of values  
string any string  

Each of the devicetype files in this folder are indexed in /config/devicetypes.xml - This file contains the additional information that defines the family of which the devicetype is a member.

Note:In the device description file the drivertype is optional. If not present it assumes Infodriver for backwards compatibility, the other options are GPI and Router.

Sample Contents - /config/devicetypes/v1647.xml

<?xml version="1.0" ?>
<!DOCTYPE device_description SYSTEM "device_description.dtd">
<device_description name="V1647" drivertype="Infodriver" >
	<param name="Comms" class="enum" style="label" slot="0"> 
		<state value="0" caption="OK" state="alarm_ok"/>
		<state value="1" caption="Fail" state="alarm_alarm"/>
	</param>
	<param name="H Pos" class="range" style="label" slot="1">
		<values min="0" max="431" step="8" bigstep="64" />
		<display inoffset="-216" multiplier="1" outoffset="" units="" dp="" />
	</param>
	<param name="H Zoom" class="range" style="label" slot="2">
		<values min="1" max="16383" step="8" bigstep="64" />
		<display inoffset="-8192" multiplier="1" outoffset="" units="" dp="" />
	</param>
	<param name="Bypass" class="enum" style="button" slot="9">
		<state value="0" caption="Normal" state="bistate_normal"/>
		<state value="1" caption="Bypass" state="bistate_warning"/>
	</param>
	<param name="Test Pattern" class="enum" style="button" slot="12">
		<state value="0" caption="OFF" state="bistate_normal"/>
		<state value="1" caption="ON" state="bistate_warning"/>
	</param>
	<param name="ARC Mode Pop" class="enum" style="popup_list" slot="54"> 
		<template id="arcmodes"/>
	</param>
	<param name="ARC Mode" class="enum" style="list" template="arcmodes" slot="54">
		<template id="arcmodes"/> 
	</param>
</device_description>

DTD

<?xml version="1.0" encoding="UTF-8"?>
	<!ELEMENT device_description (param+)>
	<!ATTLIST device_description
		name CDATA #REQUIRED
		family CDATA #IMPLIED
		drivertype CDATA #IMPLIED
		>

<!ELEMENT param ((values,display)|(state+|template))>
	<!ATTLIST param name CDATA #REQUIRED>
	<!ATTLIST param class CDATA #REQUIRED>
	<!ATTLIST param caption CDATA #REQUIRED>
	<!ATTLIST param slot CDATA #REQUIRED>

<!ELEMENT values EMPTY>
	<!ATTLIST values min CDATA #REQUIRED>
	<!ATTLIST values max CDATA #REQUIRED>
	<!ATTLIST values step CDATA #REQUIRED>
	<!ATTLIST values bigstep CDATA #REQUIRED>

<!ELEMENT display EMPTY>
	<!ATTLIST display inoffset CDATA #REQUIRED>
	<!ATTLIST display multiplier CDATA #REQUIRED>
	<!ATTLIST display outoffset CDATA #REQUIRED>
	<!ATTLIST display units CDATA #REQUIRED>
	<!ATTLIST display dp CDATA #REQUIRED>

<!ELEMENT state EMPTY>
	<!ATTLIST state value CDATA #REQUIRED>
	<!ATTLIST state caption CDATA #REQUIRED>
	<!ATTLIST state state CDATA #REQUIRED>

<!ELEMENT template EMPTY>
	<!ATTLIST template id CDATA #REQUIRED>
